## ✅ Router1（wjb001）完整配置代码

```sh
enable
configure terminal

! IKE策略配置
crypto isakmp policy 10
 encryption 3des
 hash md5
 authentication pre-share
 group 2
exit

! IKE预共享秘钥配置
crypto isakmp key wjb001wjb002 address 192.168.2.2

! 配置IPSec变换集
crypto ipsec transform-set wjb001TS esp-des

! 用ACL定义需要IPSec保护的流量
access-list 110 permit icmp 192.168.1.0 0.0.0.255 192.168.3.0 0.0.0.255

! 创建crypto map
crypto map wjb001map 10 ipsec-isakmp
 set peer 192.168.2.2
 set transform-set wjb001TS
 match address 110
exit

! 应用crypto map到路由器端口上
interface GigabitEthernet0/0/1
 crypto map wjb001map
exit

write memory
```
